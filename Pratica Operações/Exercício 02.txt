void main() {
  
  int num1 = 18;
  int num2 = 50;
  
   if(num1 > num2){
  print('Número 1 é menor');
}else{
  print("Número 2 é maior");
}
  if(num1 == num2){
  print('Números são iguais');
}else{
  print("Números são diferentes");
}
   
}
